10 _CPU_AND_GPU_CODE_ inline float computeUpdatedVoxelDepthInfo(DEVICEPTR(TVoxel) &voxel, const THREADPTR(Vector4f) & pt_model, const CONSTPTR(Matrix4f) & M_d,
11  const CONSTPTR(Vector4f) & projParams_d, float mu, int maxW, const CONSTPTR(float) *depth, const CONSTPTR(Vector2i) & imgSize)
12 {
13  Vector4f pt_camera; Vector2f pt_image;
14  float depth_measure, eta, oldF, newF;
15  int oldW, newW;
16 
17  // project point into image
18  pt_camera = M_d * pt_model;
19  if (pt_camera.z <= 0) return -1;
20 
21  pt_image.x = projParams_d.x * pt_camera.x / pt_camera.z + projParams_d.z;
22  pt_image.y = projParams_d.y * pt_camera.y / pt_camera.z + projParams_d.w;
23  if ((pt_image.x < 1) || (pt_image.x > imgSize.x - 2) || (pt_image.y < 1) || (pt_image.y > imgSize.y - 2)) return -1;
24 
25  // get measured depth from image
26  depth_measure = depth[(int)(pt_image.x + 0.5f) + (int)(pt_image.y + 0.5f) * imgSize.x];
27  if (depth_measure <= 0.0) return -1;
28 
29  // check whether voxel needs updating
30  eta = depth_measure - pt_camera.z;
31  if (eta < -mu) return eta;
32 
33  // compute updated SDF value and reliability
34  oldF = TVoxel::SDF_valueToFloat(voxel.sdf); oldW = voxel.w_depth;
35 
36  newF = MIN(1.0f, eta / mu);
37  newW = 1;
38 
39  newF = oldW * oldF + newW * newF;
40  newW = oldW + newW;
41  newF /= newW;
42  newW = MIN(newW, maxW);
43 
44  // write back
45  voxel.sdf = TVoxel::SDF_floatToValue(newF);
46  voxel.w_depth = newW;
47 
48  return eta;
49 }

118 _CPU_AND_GPU_CODE_ inline void buildHashAllocAndVisibleTypePP(DEVICEPTR(uchar) *entriesAllocType, DEVICEPTR(uchar) *entriesVisibleType, int x, int y,
119  DEVICEPTR(Vector4s) *blockCoords, const CONSTPTR(float) *depth, Matrix4f invM_d, Vector4f projParams_d, float mu, Vector2i imgSize,
120  float oneOverVoxelSize, const CONSTPTR(ITMHashEntry) *hashTable, float viewFrustum_min, float viewFrustum_max)
121 {
122  float depth_measure; unsigned int hashIdx; int noSteps;
123  Vector3f pt_camera_f, point_e, point, direction; Vector3s blockPos;
124 
125  depth_measure = depth[x + y * imgSize.x];
126  if (depth_measure <= 0 || (depth_measure - mu) < 0 || (depth_measure - mu) < viewFrustum_min || (depth_measure + mu) > viewFrustum_max) return;
127 
128  pt_camera_f.z = depth_measure;
129  pt_camera_f.x = pt_camera_f.z * ((float(x) - projParams_d.z) * projParams_d.x);
130  pt_camera_f.y = pt_camera_f.z * ((float(y) - projParams_d.w) * projParams_d.y);
131 
132  float norm = sqrt(pt_camera_f.x * pt_camera_f.x + pt_camera_f.y * pt_camera_f.y + pt_camera_f.z * pt_camera_f.z);
133 
134  Vector4f tmp;
135  tmp.x = pt_camera_f.x * (1.0f - mu / norm);
136  tmp.y = pt_camera_f.y * (1.0f - mu / norm);
137  tmp.z = pt_camera_f.z * (1.0f - mu / norm);
138  tmp.w = 1.0f;
139  point = TO_VECTOR3(invM_d * tmp) * oneOverVoxelSize;
140  tmp.x = pt_camera_f.x * (1.0f + mu / norm);
141  tmp.y = pt_camera_f.y * (1.0f + mu / norm);
142  tmp.z = pt_camera_f.z * (1.0f + mu / norm);
143  point_e = TO_VECTOR3(invM_d * tmp) * oneOverVoxelSize;
144 
145  direction = point_e - point;
146  norm = sqrt(direction.x * direction.x + direction.y * direction.y + direction.z * direction.z);
147  noSteps = (int)ceil(2.0f*norm);
148 
149  direction /= (float)(noSteps - 1);
150 
151  //add neighbouring blocks
152  for (int i = 0; i < noSteps; i++)
153  {
154  blockPos = TO_SHORT_FLOOR3(point);
155 
156  //compute index in hash table
157  hashIdx = hashIndex(blockPos);
158 
159  //check if hash table contains entry
160  bool isFound = false;
161 
162  ITMHashEntry hashEntry = hashTable[hashIdx];
163 
164  if (IS_EQUAL3(hashEntry.pos, blockPos) && hashEntry.ptr >= -1)
165  {
166  //entry has been streamed out but is visible or in memory and visible
167  entriesVisibleType[hashIdx] = (hashEntry.ptr == -1) ? 2 : 1;
168 
169  isFound = true;
170  }
171 
172  if (!isFound)
173  {
174  bool isExcess = false;
175  if (hashEntry.ptr >= -1) //seach excess list only if there is no room in ordered part
176  {
177  while (hashEntry.offset >= 1)
178  {
179  hashIdx = SDF_BUCKET_NUM + hashEntry.offset - 1;
180  hashEntry = hashTable[hashIdx];
181 
182  if (IS_EQUAL3(hashEntry.pos, blockPos) && hashEntry.ptr >= -1)
183  {
184  //entry has been streamed out but is visible or in memory and visible
185  entriesVisibleType[hashIdx] = (hashEntry.ptr == -1) ? 2 : 1;
186 
187  isFound = true;
188  break;
189  }
190  }
191 
192  isExcess = true;
193  }
194 
195  if (!isFound) //still not found
196  {
197  entriesAllocType[hashIdx] = isExcess ? 2 : 1; //needs allocation
198  if (!isExcess) entriesVisibleType[hashIdx] = 1; //new entry is visible
199 
200  blockCoords[hashIdx] = Vector4s(blockPos.x, blockPos.y, blockPos.z, 1);
201  }
202  }
203 
204  point += direction;
205  }
206 }
