31 _CPU_AND_GPU_CODE_ inline void filterDepth(DEVICEPTR(float) *imageData_out, const CONSTPTR(float) *imageData_in, int x, int y, Vector2i imgDims)
32 {

34

36  if (z < 0.0f) { imageData_out[x + y * imgDims.x] = -1.0f; return; }
37 
38  float sigma_z = 1.0f / (0.0012f + 0.0019f*(z - 0.4f)*(z - 0.4f) + 0.0001f / sqrt(z) * 0.25f);
39 
40  for (int i = -2, count = 0; i <= 2; i++) for (int j = -2; j <= 2; j++, count++)

43  if (tmpz < 0.0f) continue;
